505 static bool parseRotationMatrix(const yarp::os::Value& ini, iDynTree::Rotation& rotation)
506 {
507  if (ini.isNull() || !ini.isList())
508  {
509  return false;
510  }
511  yarp::os::Bottle *outerList = ini.asList();
512  if (!outerList || outerList->size() != 3)
513  {
514  return false;
515  }
516  for (int row = 0; row < outerList->size(); ++row)
517  {
518  yarp::os::Value& innerValue = outerList->get(row);
519  if (innerValue.isNull() || !innerValue.isList())
520  {
521  return false;
522  }
523  yarp::os::Bottle *innerList = innerValue.asList();
524  if (!innerList || innerList->size() != 3)
525  {
526  return false;
527  }
528  for (int column = 0; column < innerList->size(); ++column)
529  {
530  rotation.setVal(row, column, innerList->get(column).asDouble());
531  }
532  }
533  return true;
534 }
